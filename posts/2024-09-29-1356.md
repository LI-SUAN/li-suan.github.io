# C语言 排序

## 运行代码

```C
#include <stdio.h>
#include <math.h>

int main() {

	/* 冒泡排序 */
	int replacement_times = 0;
	int array_a[11] = {3, 6, 7, 1, 2, 0, 9, 8, 4, 5, 9};
	for (int j = 0; j < 10; j++) {
		for (int i = 0; i < 10-j; i++) {
			if (array_a[i] > array_a[i+1]) {
				array_a[i] = array_a[i] + array_a[i+1];
				array_a[i+1] = array_a[i] - array_a[i+1];
				array_a[i] = array_a[i] - array_a[i+1];

				replacement_times++;
			}
		}
        printf("第%d次运行结果: \t", j+1);
		for (int i = 0; i < 11; i++) {
			printf("%d\t", array_a[i]);
		}
		printf("\n");
	}
	printf("冒泡排序总共替换了%d次\n\n", replacement_times);



	/* 选择排序 */
	replacement_times = 0;
	int array_b[11] = {3, 6, 7, 1, 2, 0, 9, 8, 4, 5, 9};
	int min, temp;
	for (int i = 0; i < 11; i++)
	{
		min = i;
		for (int j = i+1; j < 11; j++)
		{
			if (array_b[min] < array_b[j])
			{
				min = j;
			}
		}
		temp = array_b[i];
		array_b[i] = array_b[min];
		array_b[min] = temp;

		replacement_times++;
		printf("第%d次运行结果: \t", i+1);
		for (int i = 0; i < 11; i++) {
			printf("%d\t", array_b[i]);
		}
		printf("\n");
	}
	printf("选择排序总共替换了%d次\n\n", replacement_times);



	return 0;
}
```
